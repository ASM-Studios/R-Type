115 sf::RenderWindow window(sf::VideoMode(POPUP_WIDTH, POPUP_HEIGHT), "Message");
116 sf::Font font;
117 if (!font.loadFromFile(FONT_FILENAME)) {
118 throw GuiException("Cannot load font");

120
121 sf::Text text;
122 text.setFont(font);
123 text.setString(_message);
124 text.setCharacterSize(FONT_SIZE);
125 text.setPosition(TEXT_POS, TEXT_POS);
126 text.setFillColor(sf::Color::White);
127
128 while (window.isOpen()) {
129 sf::Event event{};
130 while (window.pollEvent(event)) {
131 if (event.type == sf::Event::Closed) {
132 window.close();
133 }
134 if (event.type == sf::Event::KeyPressed) {
135 if (event.key.code == sf::Keyboard::Escape || event.key.code == sf::Keyboard::Q) {
136 window.close();
137 }
138 }
139 }
140
141 window.clear();
142 window.draw(text);
