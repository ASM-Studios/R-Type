41void BehaviorFunc::setSpriteSheetFromInput(GameLogicMode mode, const ecs::Entity& entity, float deltaTime) {
42 auto registry = entity.getRegistry();
43 auto& input = registry->getComponent<ecs::component::Input>(entity);
44 auto& sprite = registry->getComponent<ecs::component::Sprite>(entity);
45 static float upDuration = 0.0F;
46 static float downDuration = 0.0F;
47 int spriteSheetID = 2;
48
49 if (input.isFlagSet(ecs::component::Input::MoveUp) && input.isFlagSet(ecs::component::Input::MoveDown)) {
50 upDuration = std::max(0.0F, upDuration - deltaTime);
51 downDuration = std::max(0.0F, downDuration - deltaTime);
52 } else if (input.isFlagSet(ecs::component::Input::MoveUp)) {
53 upDuration += deltaTime;
54 downDuration = 0.0F;
55 } else if (input.isFlagSet(ecs::component::Input::MoveDown)) {
56 downDuration += deltaTime;
57 upDuration = 0.0F;
58 } else {
59 upDuration = 0;
60 downDuration = 0;
61 }
62
63 if (upDuration > 0.33F) {
64 spriteSheetID = 4;
65 } else if (upDuration > 0.17F) {
66 spriteSheetID = 3;
67 } else if (upDuration > 0.0F) {
68 spriteSheetID = 2;
69 }
70
71 if (downDuration > 0.33F) {
72 spriteSheetID = 0;
73 } else if (downDuration > 0.17F) {
74 spriteSheetID = 1;
75 } else if (downDuration > 0.0F) {
76 spriteSheetID = 2;
77 }
78 if (sprite.getStateID() != spriteSheetID) {
79 sprite.setStateID(spriteSheetID);

109void BehaviorFunc::handleInput(GameLogicMode mode, const ecs::Entity& entity, float timePerTick) {
110 auto registry = entity.getRegistry();
111 auto& input = registry->getComponent<ecs::component::Input>(entity);
112 if (input.inputFlags == 0) {
113 setSpriteSheetFromInput(mode, entity, timePerTick);
114 return;
115 }
116
117 auto& position = registry->getComponent<ecs::component::Position>(entity);
118 auto& lastShot = registry->getComponent<ecs::component::LastShot>(entity);
119 int16_t speed = SPEED * timePerTick;
120
121 ecs::component::Position offset(0, 0);
122 if (input.isFlagSet(ecs::component::Input::MoveLeft)) {
123 offset.x -= speed;
124 }
125 if (input.isFlagSet(ecs::component::Input::MoveRight)) {
126 offset.x += speed;
127 }
128 if (input.isFlagSet(ecs::component::Input::MoveUp)) {
129 offset.y -= speed;
130 }
131 if (input.isFlagSet(ecs::component::Input::MoveDown)) {
132 offset.y += speed;
133 }
134 position.move(offset);
135 setSpriteSheetFromInput(mode, entity, timePerTick);
136 auto currentTime = std::chrono::steady_clock::now();
137 float const deltaTimeShot = std::chrono::duration<float>(currentTime - lastShot.lastShotTime).count();
138
139 if (input.isFlagSet(ecs::component::Input::ReleaseShoot) && deltaTimeShot >= SHOOT_COOLDOWN && registry->contains<ecs::component::LastShot>(entity)) {
140 if (isPureServer(mode)) {
141 sendBullet(entity);
142 }
143 lastShot.lastShotTime = currentTime;
144 }
