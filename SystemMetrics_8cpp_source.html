57 float SystemMetrics::getCPUUsage() {
58 #ifdef _WIN32
59 static FILETIME prevSysIdle, prevSysKernel, prevSysUser;
60 FILETIME sysIdle, sysKernel, sysUser;
61
62 if (!GetSystemTimes(&sysIdle, &sysKernel, &sysUser)) {
63 return -1.0f;
64 }
65
66 if (memcmp(&prevSysIdle, &sysIdle, sizeof(FILETIME)) != 0) {
67 ULONGLONG sysIdleDiff = (sysIdle.dwLowDateTime - prevSysIdle.dwLowDateTime) +
68 ((sysIdle.dwHighDateTime - prevSysIdle.dwHighDateTime) << 32);
69 ULONGLONG sysKernelDiff = (sysKernel.dwLowDateTime - prevSysKernel.dwLowDateTime) +
70 ((sysKernel.dwHighDateTime - prevSysKernel.dwHighDateTime) << 32);
71 ULONGLONG sysUserDiff = (sysUser.dwLowDateTime - prevSysUser.dwLowDateTime) +
72 ((sysUser.dwHighDateTime - prevSysUser.dwHighDateTime) << 32);
73
74 ULONGLONG sysTotal = sysKernelDiff + sysUserDiff;
75 if (sysTotal == 0)
76 return 0.0f;
77 float cpuUsage = (sysTotal - sysIdleDiff) / (float)sysTotal * 100.0f;
78
79 prevSysIdle = sysIdle;
80 prevSysKernel = sysKernel;
81 prevSysUser = sysUser;
82
83 return cpuUsage;
84 }
85 #elif __linux__
86 static long lastTotalUser = 0, lastTotalUserLow = 0, lastTotalSys = 0, lastTotalIdle = 0;
87
88
89 std::ifstream file("/proc/stat");
90 if (!file.is_open()) {
91 std::cerr << "Failed to open /proc/stat" << std::endl;
92 return 0.0f;
93 }
94 std::string cpuLabel;
95 long user, nice, system, idle, iowait, irq, softirq, steal;
96 file >> cpuLabel >> user >> nice >> system >> idle >> iowait >> irq >> softirq >> steal;
97 file.close();
98
99 if (cpuLabel != "cpu") {
100 std::cerr << "Unexpected label in /proc/stat: " << cpuLabel << std::endl;
101 return 0.0f;
102 }
103 long totalUser = user;
104 long totalUserLow = nice;
105 long totalSys = system;
106 long totalIdle = idle;
107 long total = (totalUser - lastTotalUser) + (totalUserLow - lastTotalUserLow) +
108 (totalSys - lastTotalSys);
109 long totalIdleDiff = totalIdle - lastTotalIdle;
110 if ((total + totalIdleDiff) == 0) {
111 return 0.0f;
112 }
113
114 float cpuUsage = (total * 100.0f) / (total + totalIdleDiff);
115
116 lastTotalUser = totalUser;
117 lastTotalUserLow = totalUserLow;
118 lastTotalSys = totalSys;
119 lastTotalIdle = totalIdle;
120 return cpuUsage;
121 #endif
122 return 0.0f;
123 }
