35 inline std::string getScenarioPath(const std::size_t scenarioID) {
36 libconfig::Config cfg;
37 try {
38 cfg.readFile(SCENARIO_PATH);
39 } catch (libconfig::FileIOException& e) {
40 Logger::log(LogLevel::ERR, "I/O error while reading file: " + std::string(SCENARIO_PATH));
41 return "";
42 } catch (libconfig::ParseException& e) {
43 Logger::log(LogLevel::ERR, "Parse error at " + std::to_string(e.getLine()) + ": " + e.getError());
44 return "";
45 }
46
47 const libconfig::Setting& root = cfg.getRoot();
48 const libconfig::Setting& scenarios = root["scenarios"];
49
50 for (int i = 0; i < scenarios.getLength(); ++i) {
51 const libconfig::Setting& scenario = scenarios[i];
52 int id;
53 scenario.lookupValue("id", id);
54 if (id == scenarioID) {
55 std::string path;
56 scenario.lookupValue("path", path);
57 return path;
58 }
59 }
60
61 Logger::log(LogLevel::ERR, "Scenario ID not found: " + std::to_string(scenarioID));
62 return "";
63 }
