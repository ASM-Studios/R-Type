19void GameLogic::updateTimed()
20{
21 using clock = std::chrono::steady_clock;
22 static auto prevTime = clock::now();
23 static float tmp = 0.0F;
24
25 auto cur = clock::now();
26 float const deltaTime = std::chrono::duration<float>(cur - prevTime).count();
27 prevTime = cur;
28 tmp += deltaTime;
29
30 if (tmp >= _timePerTick) {
31 this->update();
32 tmp -= _timePerTick;
33 } else {
34 float const sleepTime = _timePerTick - tmp;
35 if (sleepTime > 0) {
36 std::this_thread::sleep_for(std::chrono::duration<float>(sleepTime));
37 tmp += sleepTime;
38 }
39 }

60 ecs::Registry& registry = ecs::RegistryManager::getInstance().getRegistry();

64 float const animation_speed = texture.getTexture(sprite.getSpriteID()).getAnimSpeed();
65 auto& tags = registry.getComponent<ecs::component::Tags>(entity);
66 animation.elapsedTime += _timePerTick;
67 if (animation.elapsedTime >= animation_speed) {
68 animation.elapsedTime -= animation_speed;

70 int const nbFrame = texture.getTexture(sprite.getSpriteID()).getFrameCount();
71 if (animation.currFrame >= nbFrame) {
72 if (tags.hasTag(ecs::component::Tag::Explosion)) {
