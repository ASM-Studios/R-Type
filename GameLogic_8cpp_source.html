58void GameLogic::updateTimed() {
59 if (!this->_isRunning) {
60 return;
61 }
62 using clock = std::chrono::steady_clock;
63 static auto prevTime = clock::now();
64 static float tmp = 0.0F;
65
66 auto cur = clock::now();
67 float const deltaTime = std::chrono::duration<float>(cur - prevTime).count();
68 prevTime = cur;
69 tmp += deltaTime;
70 _totalTime += deltaTime;
71 if (tmp >= _timePerTick) {
72 this->update();
73 tmp -= _timePerTick;
74 } else {
75 float const sleepTime = _timePerTick - tmp;
76 if (sleepTime > 0) {
77 std::this_thread::sleep_for(std::chrono::duration<float>(sleepTime));
78 tmp += sleepTime;
79 }
80 }
81}

144void GameLogic::updateAnimation(const ecs::Entity& entity) {
145 ecs::Registry& registry = ecs::RegistryManager::getInstance().getRegistry();
146 auto& animation = registry.getComponent<ecs::component::Animation>(entity);
147 auto& sprite = registry.getComponent<ecs::component::Sprite>(entity);
148 auto& texture = TextureLoader::getInstance();
149 float const animation_speed = texture.getTexture(sprite.getSpriteID()).getAnimSpeed();
150 auto& tags = registry.getComponent<ecs::component::Tags>(entity);
151 animation.elapsedTime += _timePerTick;
152 if (animation.elapsedTime >= animation_speed) {
153 animation.elapsedTime -= animation_speed;
154 animation.currFrame++;
155 int const nbFrame = texture.getTexture(sprite.getSpriteID()).getFrameCount();
156 if (animation.currFrame >= nbFrame) {
157 if (tags.hasTag(ecs::component::Tag::Explosion)) {
158 registry.removeEntity(entity);
159 return;
160 }
161 animation.currFrame = 0;
162 }
163 sprite.setStateID(animation.currFrame);
164 }
