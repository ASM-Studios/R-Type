29void GUI::SpriteManager::loopDirectory(const std::string &spritePath, const std::string &relativePath) {
30 boost::filesystem::path const path(spritePath + relativePath);
31 if (exists(path) && is_directory(path)) {
32 for (const auto& entry : boost::filesystem::directory_iterator(path)) {
33 if (is_regular_file(entry) && entry.path().extension() == ".png") {
34 std::string name = relativePath + boost::algorithm::erase_all_copy(entry.path().filename().string(), ".png");
35 importSprites(spritePath, name);
36 } else if (is_directory(entry)) {
37 std::string newRelativePath = relativePath + entry.path().filename().string() + "/";
38 loopDirectory(spritePath, newRelativePath);

40 }
41 } else {
42 std::cerr << "Directory does not exist or is not a directory: " << spritePath << std::endl;

54void GUI::SpriteManager::importSprites(const std::string& spritePath, const std::string& name) {
55 libconfig::Config cfg;
56 try {
57 cfg.readFile(_spritesConfigPath);
58 } catch ([[maybe_unused]] const libconfig::FileIOException &fioex) {
59 throw GuiException("I/O error while reading file.");
60 } catch (const libconfig::ParseException &pex) {
61 throw GuiException("Parse error at " + std::string(pex.getFile()) + ":" + std::to_string(pex.getLine()) + " - " + pex.getError());
62 }
63
64 if (cfg.exists(name)) {
65 const libconfig::Setting& setting = cfg.lookup(name);
66 int spriteID;
67 if (!setting.lookupValue("id", spriteID)) {
68 throw GuiException("ID not found for sprite: " + name);
69 }
70
71 if (_textures.contains(spriteID)) {
72 throw GuiException("Sprite sheet '" + name + "' already loaded.");
73 }
74
75 int row, col;
76 if (float scale; setting.lookupValue("row", row) && setting.lookupValue("col", col) && setting.lookupValue("scale", scale)) {
77 bool autoscale = setting.lookupValue("autoscale", autoscale) ? autoscale : false;
78 bool center = setting.lookupValue("center", center) ? center : true;
79 const auto texture = std::make_shared<sf::Texture>();
80 if (!texture->loadFromFile(spritePath + "/" + name + ".png")) {
81 throw GuiException("Failed to load texture: " + spritePath + "/" + name + ".png");
82 }
83
84 _textures[spriteID] = texture;
85
86 std::size_t const spriteWidth = texture->getSize().x / col;
87 std::size_t const spriteHeight = texture->getSize().y / row;
88
89 Logger::log(LogLevel::INFO, "Sprite sheet '" + name + "' loaded with " + std::to_string(row * col) + " sprites");
90
91 std::size_t id = 0;
92 for (int r = 0; r < row; ++r) {
93 for (int c = 0; c < col; ++c) {
94 const auto sprite = std::make_shared<sf::Sprite>();
95 sprite->setTexture(*texture);
96 sprite->setTextureRect(sf::IntRect(c * spriteWidth, r * spriteHeight, spriteWidth, spriteHeight));
97 if (autoscale) {
98 const float scaleFactorX = _width / sprite->getGlobalBounds().width;
99 const float scaleFactorY = _height / sprite->getGlobalBounds().height;
100 sprite->setScale(scaleFactorX, scaleFactorY);
101 } else { sprite->setScale(scale, scale); }
102 if (center) { sprite->setOrigin(spriteWidth / 2, spriteHeight / 2); }
103 _sprites[spriteID][id++] = sprite;
104 }
105 }
106 } else {
107 throw GuiException(name + " missing key or wrong typed value in configuration file.");
108 }
109 } else {
110 Logger::log(LogLevel::WARNING, name + " not found in configuration, skipping.");
111 }
112}
